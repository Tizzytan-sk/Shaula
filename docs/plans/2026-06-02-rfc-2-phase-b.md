## Goal

一句话：让用户在 agent 真正执行危险工具调用（`rm -rf` / `git reset --hard` 等）前能看到审批气泡，可选 allow / edit / deny，且 deny 时 agent 收到原因后能自然换策略。

## Architecture（与 RFC §3 一致）

```
当前架构（read-only）:
  AgentSession ──subscribe──→ ring buffer ──SSE──→ 前端（只能看）

目标架构（双向可干预）:
  AgentSession ──subscribe──→ ring buffer ──SSE──→ 前端
       ▲                            (推 approval_request 自定义事件)
       │
       │ 注入 inline ExtensionFactory（DefaultResourceLoader）
       │
  CollabExtension（shaula-agent 新写）
       └─ on("tool_call") → 查 rule → 命中则 await pendingApproval
                                          ↑
                                          │ resolve
                            POST /api/agent/[id]/approval
```

**关键技术决策（已验证 SDK 支持）**：
- `createAgentSession` 接受 `resourceLoader?: ResourceLoader` 参数
- `DefaultResourceLoader` 构造函数接受 `extensionFactories?: ExtensionFactory[]`
- `ExtensionAPI.on("tool_call", handler)` 的 handler 可返回 `Promise<ToolCallEventResult>`，runner 会 await
- `ToolCallEventResult = { block?: boolean; reason?: string }`
- `event.input` 可 mutate（按 SDK 文档要求 in place）

**Step 3：写 matcher.ts 让单测过**

```ts
import type { ApprovalRule, ToolCallMatcher } from "./types";
import type { ToolCallEvent } from "@earendil-works/pi-coding-agent";

export function matchRule(
  event: ToolCallEvent,
  rules: ApprovalRule[]
): ApprovalRule | undefined {
  for (const rule of rules) {
    if (matchesMatcher(event, rule.match)) return rule;
  }
  return undefined;
}

function matchesMatcher(event: ToolCallEvent, m: ToolCallMatcher): boolean {
  // toolName check
  if (m.toolName != null) {
    const names = Array.isArray(m.toolName) ? m.toolName : [m.toolName];
    if (!names.includes(event.toolName)) return false;
  }
  // inputMatch check
  if (m.inputMatch) {
    for (const [key, cond] of Object.entries(m.inputMatch)) {
      const val = (event.input as Record<string, unknown>)[key];
      if (typeof val !== "string") return false;
      if (cond.contains?.length) {
        if (!cond.contains.some((kw) => val.includes(kw))) return false;
      }
      if (cond.regex) {
        if (!new RegExp(cond.regex).test(val)) return false;
      }
    }
  }
  return true;
}
```

**Step 5：写 extension.ts（B2 版：auto-allow + console.log）**

```ts
import type { ExtensionFactory } from "@earendil-works/pi-coding-agent";
import { matchRule } from "./matcher";
import type { ApprovalRule, ApprovalRequest, ApprovalResponse } from "./types";

export interface CollabExtensionOptions {
  getRules: () => ApprovalRule[];
  onApprovalNeeded: (req: ApprovalRequest) => Promise<ApprovalResponse>;
  getAgentId: () => string;
}

export function createCollabExtension(opts: CollabExtensionOptions): ExtensionFactory {
  return (pi) => {
    pi.on("tool_call", async (event) => {
      try {
        const rule = matchRule(event, opts.getRules());

        if (!rule || rule.on === "auto-allow") return; // 放行
        if (rule.on === "auto-deny") {
          return { block: true, reason: rule.denyReason ?? "denied by rule" };
        }

        // rule.on === "ask"
        const req: ApprovalRequest = {
          id: `${opts.getAgentId()}:${event.toolCallId}`,
          agentId: opts.getAgentId(),
          toolCallId: event.toolCallId,
          toolName: event.toolName,
          input: event.input as Record<string, unknown>,
          reason: "rule",
          ruleId: rule.id,
          defaultDecision: "deny",
          createdAt: Date.now(),
        };

        const resp = await opts.onApprovalNeeded(req);

        if (resp.decision === "allow") return;
        return { block: true, reason: resp.denyReason ?? rule.denyReason ?? "denied by user" };
      } catch (err) {
        // R6：extension 抛错时默认 allow，不阻塞 agent
        console.error("[collab] extension error, defaulting allow:", err);
        return;
      }
    });
  };
}
```

**Step 1：写 server-store.ts**

```ts
// lib/collab/server-store.ts
import "server-only";
import type { ApprovalRequest, ApprovalResponse } from "./types";

interface StoreEntry {
  request: ApprovalRequest;
  resolve: (resp: ApprovalResponse) => void;
}

interface CollabRegistry {
  approvals: Map<string, StoreEntry>;
}

const g = globalThis as unknown as { __shaulaAgentCollab?: CollabRegistry };
if (!g.__shaulaAgentCollab) {
  g.__shaulaAgentCollab = { approvals: new Map() };
}
const store = g.__shaulaAgentCollab!;

export function registerPendingApproval(
  request: ApprovalRequest
): Promise<ApprovalResponse> {
  return new Promise<ApprovalResponse>((resolve) => {
    store.approvals.set(request.id, { request, resolve });
  });
}

export function resolveApproval(id: string, resp: ApprovalResponse): boolean {
  const entry = store.approvals.get(id);
  if (!entry) return false;
  store.approvals.delete(id);
  entry.resolve(resp);
  return true;
}

export function getPendingApproval(id: string): ApprovalRequest | undefined {
  return store.approvals.get(id)?.request;
}
```

```ts
// app/api/agent/[id]/approval/route.ts
import { NextRequest, NextResponse } from "next/server";
import { resolveApproval } from "@/lib/collab/server-store";

export async function POST(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  await params; // agent id 当前不用，但保留
  const body = await req.json();
  const { approvalId, decision, denyReason } = body;
  const ok = resolveApproval(approvalId, { decision, denyReason });
  if (!ok) return NextResponse.json({ error: "not found" }, { status: 404 });
  return NextResponse.json({ ok: true });
}
```

**Step 4：useApprovals hook**

```ts
// app/hooks/useApprovals.ts
export function useApprovals(agentId: string | null, messages: MessageEntry[]) {
  // messages 里找 type === "approval" 且 status === "pending" 的
  const pending = useMemo(
    () => messages.filter((m): m is ApprovalMessageEntry => m.type === "approval" && m.status === "pending"),
    [messages]
  );

  const approve = useCallback(
    async (approvalId: string) => {
      await fetch(`/api/agent/${agentId}/approval`, {
        method: "POST",
        body: JSON.stringify({ approvalId, decision: "allow" }),
      });
    },
    [agentId]
  );

  const deny = useCallback(
    async (approvalId: string, reason?: string) => {
      await fetch(`/api/agent/${agentId}/approval`, {
        method: "POST",
        body: JSON.stringify({ approvalId, decision: "deny", denyReason: reason }),
      });
    },
    [agentId]
  );

  return { pending, approve, deny };
}
```

**Step 5：扩展 chat-reducer / useAgentEvents**

- 在 `MessageEntry` 类型 union 里加 `ApprovalMessageEntry`
- `useAgentEvents` 在收到 approval_request 自定义事件时 dispatch reducer action 注入 message
- 收到该 approval id 的 resolve（前端调 POST 后没有 SSE 回声，但 status 改成 "resolved" 是前端 useApprovals 内部 fire-and-forget 后更新本地的 messages 状态？）

⚠️ **这块是 B3 最复杂的部分**：approval message 的 status 流转（pending → resolved）怎么同步前后端。

**方案**：
- approval message 的 status 在前端**乐观更新**（点 approve 立刻 setMessages 把这条 status 改为 resolved）
- server 端 resolveApproval 是单向的（不需要再推 SSE 回声）
- 如果 SSE 断线重连，approval 的 status 仍是 pending（因为 ring buffer 只记 push 那一刻）。可接受，让用户重点一次即可
  - **Phase C 优化**：server 端记 approval 的 resolved 状态，前端首次 fetch 时拉一次

```tsx
      <div className="text-sm font-semibold">⚠ agent 想运行 {request.toolName}:</div>
      <pre className="bg-neutral-950 p-2 mt-2 text-xs font-mono whitespace-pre-wrap">
        {JSON.stringify(request.input, null, 2)}
      </pre>
      {request.ruleId && (
        <div className="text-xs text-neutral-500 mt-1">匹配规则: {request.ruleId}</div>
      )}
      {status === "pending" ? (
        <div className="flex gap-2 mt-3">
          <button onClick={() => onApprove(request.id)} className="px-3 py-1 bg-green-700 rounded text-sm">Allow</button>
          <button onClick={() => onDeny(request.id)} className="px-3 py-1 bg-red-700 rounded text-sm">Deny</button>
        </div>
      ) : (
        <div className="text-xs text-neutral-500 mt-2">已 {status === "approved" ? "通过" : "拒绝"}</div>
      )}
    </div>
  );
}
```
